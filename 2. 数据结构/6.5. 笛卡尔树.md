基于单调栈的思想，我们可以实现笛卡尔树这一数据结构。

```cpp
#include <bits/stdc++.h>

using std::max, std::min;
using std::vector, std::pair, std::string, std::cin, std::cout;

#define all(x) std::begin(x), std::end(x)
#define sz(x) (int) (x).size()

using i64 = long long;
using u64 = unsigned long long;
using i128 = __int128_t;
using u128 = __uint128_t;
using pii = pair<int, int>;
using vi = vector<int>;

int main() {
    cin.tie(nullptr)->sync_with_stdio(false);
    int n;
    cin >> n;
    vi a(n + 1), ls(n + 1, 0), rs(n + 1, 0);
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    std::stack<int> st;
    st.push(1);
    for (int i = 2; i <= n; i++) {
        int cur = 0;
        while (!st.empty() && a[st.top()] > a[i]) {
            cur = st.top();
            st.pop();
        }
        ls[i] = cur; // 当前节点的左孩子一定是最后一个被弹出来的
        if (!st.empty()) {
            rs[st.top()] = i; // 如果非空，说明栈顶端的右孩子就是当前的节点。
        }
        st.push(i);
    }
    i64 l = 0, r = 0;
    for (int i = 1; i <= n; i++) {
        l ^= (1ll * i * (ls[i] + 1));
        r ^= (1ll * i * (rs[i] + 1));
    }
    cout << l << ' ' << r << '\n';
    return 0;
}
```
