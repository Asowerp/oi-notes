#主席树 

## 适用题目

需要通过不同版本的线段树来查询一个区间的某个值，比如区间第 $k$ 小，我们单独用线段树很难去评判。比如我们要查询 $[l,r]$ 的线段树，我们按顺序插入线段树，就可以通过 $l-1$ 版本 - $r$ 版本去求解。

## 单点修改主席树

我们期待，每次进行一次修改操作，就生成新的线段树。
 - 每次生成新的线段树，旧的节点复用，新的节点新建。空间保证是 $O(\log n)$ 的。
 - 单点修改，所以不用管 lazy_tag 的事情

### 单点修改单点查询

以下面这道题为例。

>[!example] P3919
>你需要维护这样的一个长度为 $N$ 的数组，支持如下两种操作：
>1. 在某个历史版本上修改某一个位置上的值。
>2. 访问某个历史版本上的某一位置的值。
>
>此外，每进行一次操作，就会生成一个新的版本。版本编号即为当前操作的编号（从 $1$ 开始编号，版本 $0$ 表示初始状态数组）。
>
>**对于操作 $2$，即为生成一个完全一样的版本，不作任何改动**。即，询问生成的版本是询问所访问的那个版本的复制。

![[Pasted image 20251130171436.png]]

因为大量的节点都是重复利用的，因此每次新建的空间实质上只有 $O(\log n)$。因为每个版本修改的时候实质上都经过了头节点，所以可以建立一个从版本到头节点 cnt 的满射，这也就实现了不同版本的区分。

那理解了思想之后，代码其实就很简单了。代码如下。

```cpp
// Code for P3919

#include <bits/stdc++.h>

using namespace std;

const int MAXN = 1000000 + 5;

int node_cnt = 0;

int a[MAXN];

struct Node {
    int val;
    int left, right;
} tree[MAXN * 20 * 4];

int build(int l, int r) {
    int p = ++node_cnt;
    if (l == r) {
        tree[p].val = a[l];
        return p;
    }
    int mid = l + r >> 1;
    tree[p].left = build(l, mid), tree[p].right = build(mid + 1, r);
    return p;
}

int modify(int pos, int val, int l, int r, int pre) {
    int p = ++node_cnt;
    if (l == r) {
        tree[p].val = val;
        return p;
    }
    int mid = l + r >> 1;
    tree[p].left = tree[pre].left, tree[p].right = tree[pre].right;
    if (pos <= mid) {
        tree[p].left = modify(pos, val, l, mid, tree[pre].left);
    } else {
        tree[p].right = modify(pos, val, mid + 1, r, tree[pre].right);
    }
    return p;
}

int query(int pos, int l, int r, int p) {
    if (l == r) { return tree[p].val; }
    int mid = l + r >> 1;
    if (pos <= mid) {
        return query(pos, l, mid, tree[p].left);
    } else {
        return query(pos, mid + 1, r, tree[p].right);
    }
}

int root[MAXN];

int main() {
    cin.tie(nullptr)->sync_with_stdio(false);
    cout.tie(nullptr)->sync_with_stdio(false);
    int n, m;
    cin >> n >> m;
    for (int i = 1; i <= n; i++) cin >> a[i];
    root[0] = build(1, n);
    for (int i = 1; i <= m; i++) {
        int v, op;
        cin >> v >> op;
        if (op == 1) {
            int p, c;
            cin >> p >> c;
            root[i] = modify(p, c, 1, n, root[v]);
        } else {
            int p;
            cin >> p;
            root[i] = root[v];
            cout << query(p, 1, n, root[v]) << endl;
        }
    }
    return 0;
}

```

### 单点修改范围查询

以下面这题为例。

>[!example] P3834
>给定 $n$ 个整数构成的序列 $a$，将对于指定的闭区间 $[l, r]$ 查询其区间内的第 $k$ 小值。

我们这样建立不同版本的线段树：
 - 第 $v$ 个版本的线段树，代表从 $a[1..v]$ 中不同数目出现的次数。
因此，我们要查询 $[l,r]$ 的第 $k$ 小，只用取出第 $l-1$ 个版本和第 $r$ 个版本去查询就好了。

```cpp
// Code for P3834

#include <bits/stdc++.h>

using namespace std;

const int MAXN = 2e5 + 5;

struct Node {
    int val;
    int l, r;
} tree[MAXN * 20];

int node_cnt;
int root[MAXN];

int add(int pos, int val, int l, int r, int pre) {
    int p = ++node_cnt;
    tree[p] = tree[pre];
    if (l == r) {
        tree[p].val += val;
        return p;
    }
    int mid = l + r >> 1;
    if (pos <= mid)
        tree[p].l = add(pos, val, l, mid, tree[pre].l);
    else
        tree[p].r = add(pos, val, mid + 1, r, tree[pre].r);
    tree[p].val = tree[tree[p].l].val + tree[tree[p].r].val;
    return p;
}

int queryKth(int l_root, int r_root, int l, int r, int k) {
    // 第 l_root 版本的树和 r_root 版本的树做差；
    // 同时进左 / 右子树！
    if (l == r) return l;
    int mid = l + r >> 1;
    int cnt = tree[tree[r_root].l].val - tree[tree[l_root].l].val;
    if (cnt >= k) {
        return queryKth(tree[l_root].l, tree[r_root].l, l, mid, k);
    } else {
        return queryKth(tree[l_root].r, tree[r_root].r, mid + 1, r,
                        k - cnt);
    }
}

int n, m;

int main() {
    cin >> n >> m;
    vector<int> a(n);
    for (int i = 0; i < n; i++) { cin >> a[i]; }
    vector<int> sorteda = a;
    sort(sorteda.begin(), sorteda.end());
    sorteda.erase(unique(sorteda.begin(), sorteda.end()),
                  sorteda.end());
    int k = sorteda.size();
    vector<int> idx(k + 1);
    for (int i = 1; i <= k; i++) { idx[i] = sorteda[i - 1]; }
    for (int i = 0; i < n; i++) {
        a[i] = lower_bound(sorteda.begin(), sorteda.end(), a[i]) -
               sorteda.begin() + 1;
        root[i + 1] = add(a[i], 1, 1, n, root[i]);
    }
    for (int i = 1; i <= m; i++) {
        int l, r, k_val;
        cin >> l >> r >> k_val;
        cout << idx[queryKth(root[l - 1], root[r], 1, n, k_val)]
             << endl;
    }
    return 0;
}
```
