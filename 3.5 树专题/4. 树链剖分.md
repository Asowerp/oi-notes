# 重链剖分

重链剖分（Heavy Light Decomposition，HLD）是树链剖分最常见的形式。

我们从 [模板题](https://www.luogu.com.cn/problem/P3384) 入手讲解一下这个知识点。

为了实现重链剖分，每个节点需要维护一下信息：
 - `fa`：父节点$\newcommand{\fa}{\mathrm{fa}}$；
 - `dep`：深度$\newcommand{\dep}{\mathrm{dep}}$；
 - `siz`：子树大小$\newcommand{\siz}{\mathrm{siz}}$；
 - **`son`：重儿子$\newcommand{\son}{\mathrm{son}}$，也就是子树规模最大的儿子。若有多个，则任选其一；**
 - **`top`：所在重链的头节点$\newcommand{\top}{\mathrm{top}}$；**
 - `dfn`：递归序 / DFS 序（Depth-First search Number）的编号$\newcommand{\dfn}{\mathrm{dfn}}$；
 - `seg`：$\newcommand{\seg}{\mathrm{seg}}$$\seg_i=j$ 的含义是 $\dfn$ 序号为 $i$ 的节点原始节点编号为 $j$。


![[Pasted image 20260126220040.png]]

比如如图是一棵树。树剖的过程需要跑两次 dfs。

第一次 dfs，我们设置 $\fa, \dep, \siz, \son$。这些根据定义代码是容易编写的。

第二次 dfs，我们去设置 $\top, \dfn, \seg$。
 - 我们首先访问 $\son$ 然后设置 $\dfn$。比如如图我们会先访问 $a\to c\to g\to m\to q\to s$ 这个链，一次设置 $\dfn$ 为 1，2，...。我们把这个叫做**重链**。一个节点和它的重儿子一定属于一个重链。同时，我们设置 $\top_a=a$，且它的重儿子也是 $\top=a$。
 - 接下来我们访问轻儿子。回退到第一个有其他儿子的节点 $m$，现在就是 $m\to p$。这就开始了一个新的重链，于是设置 $\top_p=p$ 与 $\dfn_p=7$。
 - 退回到 $g$，有 $g\to l$，又开始新的重链，设置 $\top_l=l$ 与 $\dfn_l=8$，接着遍历 $\son$。
 - 然后回退到 $c$，以此类推。
 - 下面回到 $a$。这时候有两个儿子，但是都是轻儿子，**轻儿子的遍历顺序不重要。** 接下来重复上面的过程就好。
 - 在上面设置 $\dfn$ 的过程中，同步设置 $\seg$ 即可。

最后会得到这样的树（每个圈起来的都是一个重链）：
![[Pasted image 20260126223756.png]]

这样的剖分满足下面的性质：
 - 同一条链上，$\dfn$ 序号连续。
 - 同一个子树内，所有子树的节点序号一定是从 $\top$ 的 $\dfn$ 开始的 $\siz_\top$ 个，进而也是连续的。

回顾我们想要维护的操作：
- `1 x y z`，表示将树从 $x$ 到 $y$ 结点最短路径上所有节点的值都加上 $z$。
- `2 x y`，表示求树从 $x$ 到 $y$ 结点最短路径上所有节点的值之和。
- `3 x z`，表示将以 $x$ 为根节点的子树内所有节点值都加上 $z$。
- `4 x`，表示求以 $x$ 为根节点的子树内所有节点值之和。

我们上面连续的操作给了我们非常好的维护 3、4 以及子树上的信息的方法——[[4. 线段树]]！我们完全可以构建一个以 $\dfn$ 序号为下标的线段树，然后根据 $\dfn,\seg$ 进行来回映射。

那么 1、2 等路径上的信息该怎么做呢？比如我要查询/修改 $r\to n$ 的路径信息，这肯定是牵扯到重链的跳转了。

显然 $\top_r=l$，$\top_n=h$。这说明 $r,n$ 分属不同重链。此时我们需要跳转，同时修改影响重链的部分。跳转的规则是，谁的 $\top$ 的深度大谁跳转。这里 $\dep_r>\dep_n$，因此跳转 $r$。从 $r\to l$，这一段所有的都修改 / 查询，同时跳到 $\fa_{\top_r}=g$。也就是线段树上 $\dfn_r=8$ 到 $\dfn_{\top_r}=10$ 进行修改 / 查询。再次查询。$\top_g=a$，依然不同。此时跳转 $n\leftarrow\fa_{\top_n}=c$，同时对 11 - 12 进行修改。此时 $\top_c=a=\top_g$。此时对 $\dfn_c$ 到 $\dfn_g$ 进行修改 / 查询即可。

显然，子树操作的复杂度是 $O(\log n)$，这是线段树的复杂度保证的。路径上的需要不断跳跃，这个操作复杂度是 $O(\log^2n)$，为什么呢？这个可以在 [[5. 启发式合并]] 中再获得更深刻的理解。但是下面给出了一个非常形象的证明。

>[!proof]- 证明
>
>思考下面这个形象的场景：假设 A 位于一个团体。一开始团体内只有 A。一旦遇到了一个更大的团体就合并，同时 A 会大喊一声。
>
>假设一共有 $n$ 个人，那么 A 最多喊多少声？答案是 $\log_2n$。这是因为每一个团体与 A 相遇的时候，假设是第 $k$ 次，则这个团体至少有 $2^k$ 个人。因此满足 $\sum 2^k\simeq n$，这意味着 $k\simeq\log n$。
>
>这和重链剖分有什么联系呢？现在假设我要查询的是 $a\to x$，我们只用看我们要跳多少次。现在我们要往上找 $b=\fa_{\top_a}$，这是因为 $b$ 一定有其他的重儿子，也就是有一个更大的团体。这等同于上面的 A 喊了一声。这时候跳到了 $b$，相当于和并完成了。
>
>我们这样跳下去，最多跳回到根节点，也就是上面的总共有 $n$ 个人的情况。也就是说至多跳 $\log n$ 次边我们就跳回去了。这也就是说，任何一个点到根至多只会经过 $\log n$ 个轻边。而每次跳跃都附带一个 $\log n$ 的查询 / 修改，因此路径操作的复杂度就是 $O(\log^2n)$，并且常数非常小（因为跑不满重链）。

代码如下：

```cpp
#include <bits/stdc++.h>

using namespace std;

const int MAXN = 1e5 + 5;

int n, m, r, mod, d;
vector<int> e[MAXN];
int fa[MAXN], dep[MAXN], son[MAXN], siz[MAXN], top[MAXN], dfn[MAXN],
    seg[MAXN];
int v[MAXN];

void dfs1(int u, int pa) {
    int mx = 0;
    fa[u] = pa;
    dep[u] = dep[pa] + 1;
    siz[u] = 1;
    for (int v : e[u]) {
        if (v == pa) continue;
        dfs1(v, u);
        siz[u] += siz[v];
        if (siz[v] > mx) mx = siz[v], son[u] = v;
    }
}

void dfs2(int u, int t) {
    top[u] = t;
    dfn[u] = ++d;
    seg[d] = u;
    if (!son[u]) return;
    dfs2(son[u], t);
    for (int v : e[u]) {
        if (v == fa[u] || v == son[u]) continue;
        dfs2(v, v);
    }
}

int s[MAXN << 2], lazy[MAXN << 2];

#define lson(x) (x << 1)
#define rson(x) (x << 1 | 1)

void build(int l, int r, int p) {
    if (l == r) {
        s[p] = v[seg[l]] % mod;
        return;
    }
    int mid = l + r >> 1;
    build(l, mid, lson(p)), build(mid + 1, r, rson(p));
    s[p] = (s[lson(p)] + s[rson(p)]) % mod;
}

void push_down(int l, int r, int p) {
    if (lazy[p]) {
        int mid = l + r >> 1;
        s[lson(p)] = (s[lson(p)] + (mid - l + 1) * lazy[p]) % mod,
        s[rson(p)] = (s[rson(p)] + (r - mid) * lazy[p]) % mod;
        lazy[lson(p)] = (lazy[lson(p)] + lazy[p]) % mod,
        lazy[rson(p)] = (lazy[rson(p)] + lazy[p]) % mod;
    }
    lazy[p] = 0;
}

void modify(int ml, int mr, int val, int l, int r, int p) {
    if (ml <= l && r <= mr) {
        lazy[p] = (lazy[p] + val) % mod;
        s[p] = (s[p] + (r - l + 1) * val) % mod;
        return;
    }
    push_down(l, r, p);
    int mid = l + r >> 1;
    if (ml <= mid) modify(ml, mr, val, l, mid, lson(p));
    if (mid < mr) modify(ml, mr, val, mid + 1, r, rson(p));
    s[p] = (s[lson(p)] + s[rson(p)]) % mod;
}

int query(int ql, int qr, int l, int r, int p) {
    if (ql <= l && r <= qr) { return s[p]; }
    push_down(l, r, p);
    int mid = l + r >> 1, ret = 0;
    if (ql <= mid) ret = (ret + query(ql, qr, l, mid, lson(p))) % mod;
    if (mid < qr)
        ret = (ret + query(ql, qr, mid + 1, r, rson(p))) % mod;
    return ret;
}

int main() {
    cin.tie(nullptr)->sync_with_stdio(false);
    cin >> n >> m >> r >> mod;
    for (int i = 1; i <= n; i++) { cin >> v[i]; }
    for (int i = 1; i < n; i++) {
        int u, v;
        cin >> u >> v;
        e[u].emplace_back(v), e[v].emplace_back(u);
    }
    dfs1(r, 0);
    dfs2(r, r);
    build(1, n, 1);
    for (int i = 1; i <= m; i++) {
        int op;
        cin >> op;
        if (op == 1) {
            int x, y, z;
            cin >> x >> y >> z;
            while (top[x] != top[y]) {
                if (dep[top[x]] < dep[top[y]]) swap(x, y);
                modify(dfn[top[x]], dfn[x], z, 1, n, 1);
                x = fa[top[x]];
            }
            if (dfn[x] > dfn[y]) swap(x, y);
            modify(dfn[x], dfn[y], z, 1, n, 1);
        } else if (op == 2) {
            int x, y;
            cin >> x >> y;
            int ret = 0;
            while (top[x] != top[y]) {
                if (dep[top[x]] < dep[top[y]]) swap(x, y);
                ret =
                    (ret + query(dfn[top[x]], dfn[x], 1, n, 1)) % mod;
                x = fa[top[x]];
            }
            if (dfn[x] > dfn[y]) swap(x, y);
            ret = (ret + query(dfn[x], dfn[y], 1, n, 1)) % mod;
            cout << ret << endl;
        } else if (op == 3) {
            int x, z;
            cin >> x >> z;
            modify(dfn[x], dfn[x] + siz[x] - 1, z, 1, n, 1);
        } else {
            int x;
            cin >> x;
            cout << query(dfn[x], dfn[x] + siz[x] - 1, 1, n, 1)
                 << endl;
        }
    }
    return 0;
}
```

当然，聪明的你肯定意识到，重链剖分也能解决 [[4. 树上倍增·最近公共祖先（LCA）]] 问题。如前所述，单次查询时间复杂度也是 $O(\log n)$，初始化 $O(n)$。

# 树剖中的换根操作

以 [P3979](https://www.luogu.com.cn/problem/P3979) 为例。事实上，更换 root 对路径查询没有任何影响，唯一有影响的是子树查询。

所以，我们完全可以不用理会换根操作，只在真正影响的时候去查询当前 root，并做对应的调整即可。（$\mathrm{LCA}$ 操作也类似进行分类讨论）

接下来就是分类讨论。以最小值为例。

假设我们已经以 1 为 root 实现了重链剖分，我们现在给定一个 $r$ 作为根，需要查询 $x$ 为头节点的子树中的最小值。怎么办呢？我们只用看 $r$ 和 $x$ 的关系即可。
1. $r=x$，这时候就相当于整棵树的最小值。
2. $r\notin T_x$，这里 $T_x$ 表示 $x$ 为头节点的子树（以 1 为头节点）。此时换根前后 $T_x$ 没有发生改变，因此接着输出 $\min\{T_x\}$ 即可。
3. $r\in T_x$，不难发现此时就是除了 $r$ 所在子树以外所有点。现在问题转化为如何查询 $r$ 所在的子树是 $x$ 的哪一个。——这实际上就是重链不断跳转。
	1. 如果 $\top[r]=\top[x]$，这说明 $r$ 位于 $\son[x]$ 的头节点；
	2. 反之，不断跳跃直到 $\top[r]=\top[x]$，最后一个 $\top$ 就是所在的子树。

以上的时间复杂度至多 $O(\log n)$。

# 边权转点权

我们现在假设要批量修改边权，我们要怎么做呢？答案是——下放到点权。

我们把边权下放到深度更深的下方节点，变成下方节点的点权之后跑重链剖分，这样我们就可以实现修改了。（这也是最自然的，因为一个节点只有一个父亲，但是可能有多个儿子，所以自然想要维护向上的那一条边；上放的话也要维护列表了，不够高效）但是，我们在修改的时候，需要忽略头节点 / LCA 的点权。这也是容易理解的。

[模板题](https://www.luogu.com.cn/problem/P3038)。

```cpp
#include <bits/stdc++.h>

using namespace std;

const int MAXN = 1e5 + 5;

int n, m;
int fa[MAXN], son[MAXN], siz[MAXN], top[MAXN], dep[MAXN], dfn[MAXN],
    seg[MAXN], d;

vector<int> e[MAXN];

void dfs1(int u, int pa) {
    fa[u] = pa;
    siz[u] = 1;
    dep[u] = dep[pa] + 1;
    for (int v : e[u]) {
        if (v == pa) continue;
        dfs1(v, u);
        siz[u] += siz[v];
        if (siz[v] > siz[son[u]]) son[u] = v;
    }
}

void dfs2(int u, int t) {
    top[u] = t;
    dfn[u] = ++d;
    seg[d] = u;
    if (!son[u]) return;
    dfs2(son[u], t);
    for (int v : e[u]) {
        if (v == fa[u] || v == son[u]) continue;
        dfs2(v, v);
    }
}

#define lson(x) (x << 1)
#define rson(x) (x << 1 | 1)

int s[MAXN << 2], lazy[MAXN << 2];

void push_down(int l, int r, int p) {
    if (lazy[p]) {
        int mid = l + r >> 1;
        s[lson(p)] += (mid - l + 1) * lazy[p],
            s[rson(p)] += (r - mid) * lazy[p];
        lazy[lson(p)] += lazy[p], lazy[rson(p)] += lazy[p];
    }
    lazy[p] = 0;
}

void modify(int ml, int mr, int l, int r, int p, int val = 1) {
    if (ml <= l && r <= mr) {
        s[p] += (r - l + 1) * val, lazy[p] += val;
        return;
    }
    push_down(l, r, p);
    int mid = l + r >> 1;
    if (ml <= mid) modify(ml, mr, l, mid, lson(p), val);
    if (mid < mr) modify(ml, mr, mid + 1, r, rson(p), val);
    s[p] = s[lson(p)] + s[rson(p)];
}

int query(int pos, int l, int r, int p) {
    if (l == r) { return s[p]; }
    push_down(l, r, p);
    int mid = l + r >> 1;
    if (pos <= mid)
        return query(pos, l, mid, lson(p));
    else
        return query(pos, mid + 1, r, rson(p));
}

int main() {
    cin.tie(nullptr)->sync_with_stdio(false);

    cin >> n >> m;
    for (int i = 1; i < n; i++) {
        int u, v;
        cin >> u >> v;
        e[u].emplace_back(v), e[v].emplace_back(u);
    }
    dfs1(1, 0);
    dfs2(1, 1);
    for (int i = 1; i <= m; i++) {
        char op;
        int u, v;
        cin >> op >> u >> v;
        if (op == 'P') {
            while (top[u] != top[v]) {
                if (dep[top[u]] < dep[top[v]]) swap(u, v);
                modify(dfn[top[u]], dfn[u], 1, n, 1);
                u = fa[top[u]];
            }
            if (dfn[u] > dfn[v]) swap(u, v);
            modify(dfn[u] + 1, dfn[v], 1, n, 1);
        } else {
            if (u != fa[v]) swap(u, v);
            cout << query(dfn[v], 1, n, 1) << endl;
        }
    }

    return 0;
}
```

# 长链剖分

长链剖分最常用的是用来优化 dp。

回顾重链剖分，我们是通过 size 来区分轻重儿子。类似，长链剖分就是通过子树高度去区分长短儿子。

类似地，我们通过两次 dfs 去分别确定两类参数。

![[Pasted image 20260129163626.png]]

比如这里，带圈的就属于一个长链。

我们类似，第一次 dfs 更新 fa, len, son。第二次更新其余的。和重链剖分唯一不同的地方仅有 siz 变成了 len。

但是，长链剖分的一个很不同的性质是，从任何一个节点往回跳，跳到根节点的时间复杂度最多是 $O(\sqrt n)$。

>[!proof]- 证明
>我们用类似的思路去思考。如果一个节点跳一次没有到根节点，说明另外一个子树比这个至少多 1。因此最劣情况下，总结点数应该是 $1+2+\cdots+k\sim k^2$，其中 $k$ 表示最劣跳跃次数。因此 $k^2\sim n$，这表明需要跳跃 $O(\sqrt n)$ 次。
>
>一个典型的最劣场景如下图所示。
>![[Pasted image 20260129172445.png]]

 在长链剖分中，如果对每个头节点都建立子树高度相关的信息，那时间复杂度是 $O(n)$ 的。

长链剖分的一个应用是 [P5903 树上 k 级祖先](https://www.luogu.com.cn/problem/P5903)。这一题要求预处理时间复杂度 $O(n\log n)$，查询复杂度 $O(1)$。因此，我们希望一次跳跃之后就能够直接查表得出答案。倍增将时间浪费在了跳的过程中。非常不牛。有没有这样一种方法，我们先利用倍增思想跳出最开始那最大的一步，后面的一步搞定呢？我们首先注意到，

>一个节点的 $k$ 级祖先所在的长链长大于等于 $k$。

基于这个思想，我们跳一步之后就可以直接查询了。头节点总是非常优秀的，因此我们期望对每个长链头节点维护 up/down 数组，表示从这个开始，往上 / 往下跳跃 $k$ 步能够到达的节点编号。我们可以空间优化为一维。
 - 对 down，这是显然的，我们可以建立一个 dfn 到数组位置的一一映射，从头节点往后的 len 都是这个长链上的，也就是说 down 实际上就是我们之前定义过的 seg 数组。
 - 对 up，我们规定，只考虑 len 范围内的。

所以，在这样的 up/down 设计下，我们需要保证第一次跳跃之后，所在的链的长度是大于等于剩余部分的。因此，记 $h=\lfloor\log k\rfloor$，我们期望跳 $2^h$，这样剩余的部分肯定满足 $k-2^h<2^h$，而跳 $2^h$ 到达的节点所在的的长链长一定大于等于 $2^h$，这样我们可以跳到 top 之后看是否还要往上/往下跳跃。不妨假设这个新结点在所在的重链的深度为 $d$，那么这个第二次的跳跃步长肯定是小于 $\min\{k-2^h,d\}\leq\mathrm{len}[\top]$ 的（**这是最精妙的地方！** 因为如果第 $k$ 级祖先在从这里到 top 的路径上，那么小于 $d$；如果在 top 顶上，那么肯定比 $k-2^h$ 小（因为这一段少了一个 $d$ 了），从而小于 $2^h$，也就是小于 $\mathrm{len}[\top]$），进而根据前面所述的 up/down 一次查询得到答案。

这样我们就做到了 $O(n\log n)$ 预处理（瓶颈在 fa 的 ST 表），$O(1)$ 查询。非常牛。

有人可能反驳：你的询问里面是不是要有 $O(\log s)$ 的查询最高位？但事实是现代 C++ 有 `__builtin_clz` 的函数能够做到 $O(1)$ 查询。所以事实上就是常数级别的。

现在的关键在于，长链剖分具体要怎么实现。我们可以学习下面的这个代码。

```cpp
#include <bits/stdc++.h>

using namespace std;

#define ui unsigned int
ui s;

inline ui get(ui x) {
    x ^= x << 13;
    x ^= x >> 17;
    x ^= x << 5;
    return s = x;
}

const int MAXN = 5e5 + 5;

int n, q;
int fa[20][MAXN], son[MAXN], dep[MAXN], len[MAXN], dfn[MAXN],
    top[MAXN], up[MAXN], down[MAXN], d;
int r;
long long ans, res;
vector<int> e[MAXN];

void dfs1(int u, int pa) {
    // len[u]=1;
    fa[0][u] = pa;
    dep[u] = dep[pa] + 1;
    for (int i = 1; i <= 19; i++) {
        fa[i][u] = fa[i - 1][fa[i - 1][u]];
    }
    for (int v : e[u]) {
        if (v == pa) continue;
        dfs1(v, u);
        if (len[v] > len[son[u]]) son[u] = v;
    }
    len[u] = len[son[u]] + 1;
}

void dfs2(int u, int t) {
    dfn[u] = ++d, down[d] = u, top[u] = t;
    if (t == u)
        up[d] = u;
    else { up[d] = fa[0][up[d - 1]]; }
    if (!son[u]) return;
    dfs2(son[u], t);
    for (int v : e[u]) {
        if (v == fa[0][u] || v == son[u]) continue;
        dfs2(v, v);
    }
}

#define count(x) (31 - __builtin_clz(x))

int main() {
    cin.tie(nullptr)->sync_with_stdio(false);

    cin >> n >> q >> s;
    for (int i = 1; i <= n; i++) {
        int f;
        cin >> f;
        if (f)
            e[i].emplace_back(f), e[f].emplace_back(i);
        else
            r = i;
    }
    dfs1(r, 0);
    dfs2(r, r);
    for (int i = 1; i <= q; i++) {
        int x = (get(s) ^ ans) % n + 1, k = (get(s) ^ ans) % dep[x];
        // cout << x << ' ' << k << endl;
        if (!k) {
            ans = x;
        } else {
            x = fa[count(k)][x], k -= (1 << count(k));
            int d = dep[x] - dep[top[x]];
            if (d == k) {
                ans = top[x];
            } else if (d < k) {
                ans = up[dfn[top[x]] + k - d];
            } else
                ans = down[dfn[top[x]] + d - k];
        }
        res ^= (i * ans);
    }
    cout << res << endl;
    return 0;
}
```

长链剖分也可以不仅仅拘泥于“长度”作为长儿子。下面这道题就是一个非常好的例子。

[BZOJ3252 攻略](https://www.luogu.com.cn/problem/P10641)：给定一个有 $n$ 个结点的树，树有点权且点权为正整数。现选取 $k$ 条从根结点出发到叶子结点的简单路径，求这些路径的并集上所有结点的点权之和的最大值。$1\leq n\leq 2\times 10^5$。

这题只用把上面的长度更换为点权就可以了。简单思考发现只用取前 $k$ 个最大的链顶即可。因为如果一个短链权值更大，那同一个 $\top$ 下另外一个长链必然更大——这是因为我们按照点权之和确定的长短儿子。

因此我们就做到了 $O(n\log n)$ 的做法。瓶颈在排序（或者用堆）。

```cpp
#include <bits/stdc++.h>

using namespace std;

#define int long long

const int MAXN = 2e5 + 5;

int n, k;
int fa[MAXN], dep[MAXN], son[MAXN], top[MAXN], dfn[MAXN], val[MAXN],
    d;
int w[MAXN];
priority_queue<int> pq;
vector<int> e[MAXN];

void dfs1(int u, int pa) {
    fa[u] = pa, dep[u] = dep[pa] + 1;
    for (int v : e[u]) {
        if (v == pa) continue;
        dfs1(v, u);
        if (val[v] > val[son[u]]) son[u] = v;
    }
    val[u] = val[son[u]] + w[u];
}

void dfs2(int u, int t) {
    if (u == t) pq.push(val[u]);
    dfn[u] = ++d, top[u] = t;
    if (!son[u]) return;
    dfs2(son[u], t);
    for (int v : e[u]) {
        if (v == son[u] || v == fa[u]) continue;
        dfs2(v, v);
    }
}

signed main() {
    cin.tie(nullptr)->sync_with_stdio(false);
    cin >> n >> k;
    for (int i = 1; i <= n; i++) cin >> w[i];
    for (int i = 1; i < n; i++) {
        int u, v;
        cin >> u >> v;
        e[u].emplace_back(v), e[v].emplace_back(u);
    }
    dfs1(1, 0), dfs2(1, 1);
    int ans = 0;
    while (!pq.empty() && k--) {
        ans += pq.top();
        pq.pop();
    }
    cout << ans << endl;
    return 0;
}
```

## 用长链剖分优化 dp

一个经典题是 [CF1009F](https://codeforces.com/problemset/problem/1009/F)。我们显然期望设计如下的 dp 来表示距离为 $x$ 的节点个数：
$$
dp[u][x]=\sum_{\fa[v]=u}dp[v][x-1]
$$
显然用类似前面 down 的设计思路，我们可以直接利用类似 $dp[\dfn[u]]$ 的方式来储存。

因此，我们发现，从长儿子转移到父节点显然是 $O(1)$。

我们现在讨论轻儿子到父节点的转移。考虑暴力枚举。假设当前子树规模是 $N$，总共有 $K$ 个子树，每个子树的长度是 $d_i$，那么这样暴力转移的时间复杂度是 $O\left(\sum d_i\right)=O(N)$。因为长链剖分是不重复的，因此总的时间复杂度就是 $O(\sum N_i)=O(n)$。加上每个节点扫描一遍获得最大的 $d_{x,k}$，总的时间复杂度依旧是 $O(n)$，足够通过此题。

可以发现，长链剖分优化 DP 的实现方式就是，长链剖分后，在维护信息的过程中，先 $O(1)$ 继承重儿子的信息，再暴力合并其余轻儿子的信息。

在统计以深度为下标的时候，重链剖分优化一般可以到 $O(n\log⁡ n)$，因为暴力统计**轻边 $\log⁡ n$ 条**，而长链剖分可以到 $O(n)$，因为**所有子树**被作为轻儿子暴力统计的代价和是 $O(n)$ 的。而被作为重儿子统计的代价，因为父亲直接继承了它的数组，所以每个点是 $O(1)$ 的，最终就是 $O(n)$。[提交记录](https://codeforces.com/problemset/submission/1009/360597430)

```cpp
#include <bits/stdc++.h>

using namespace std;

const int MAXN = 1e6 + 5;

int n;
int fa[MAXN], top[MAXN], len[MAXN], son[MAXN], dep[MAXN], dfn[MAXN],
    dp[MAXN], d;
int head[MAXN], nxt[MAXN << 1], to[MAXN << 1], ecnt;
int ans[MAXN];

inline void add_edge(int u, int v) {
    nxt[++ecnt] = head[u], to[ecnt] = v, head[u] = ecnt;
}

inline void dfs1(int u, int pa) {
    fa[u] = pa, dep[u] = dep[pa] + 1;
    for (int i = head[u]; i; i = nxt[i]) {
        int v = to[i];
        if (v == pa) continue;
        dfs1(v, u);
        if (len[v] > len[son[u]]) son[u] = v;
    }
    len[u] = len[son[u]] + 1;
}

// inline void dfs2(int u, int t) {
//     top[u] = t;
//     if (!son[u]) return;
//     dfs2(son[u], t);
//     for (int i = head[u]; i; i = nxt[i]) {
//         int v = to[i];

//         dfs2(v, v);
//     }
// }

inline void dfs3(int u) {
    dfn[u] = ++d, dp[dfn[u]] = 1;
    if (!son[u]) {
        ans[u] = 0;
        return;
    }
    dfs3(son[u]);
    ans[u] = ans[son[u]] + 1;
    int mx = dp[dfn[u] + ans[u]];
    for (int i = head[u]; i; i = nxt[i]) {
        int v = to[i];
        if (v == fa[u] || v == son[u]) continue;
        dfs3(v);
        for (int j = len[v]; j >= 1; j--) {
            dp[dfn[u] + j] += dp[dfn[v] + j - 1];
            if (dp[dfn[u] + j] >= mx) mx = dp[dfn[u] + j], ans[u] = j;
        }
    }
    if (dp[dfn[u]] >= mx) ans[u] = 0;
}

int main() {
    cin.tie(nullptr)->sync_with_stdio(false);
    cin >> n;
    for (int i = 1; i < n; i++) {
        int u, v;
        cin >> u >> v;
        add_edge(u, v), add_edge(v, u);
    }
    dfs1(1, 0), dfs3(1);
    for (int i = 1; i <= n; i++) cout << ans[i] << endl;
    return 0;
}
```

也就是说，这种树上 dp 问题，只要 $dp[u][x]$ 中的 $x$ 和高度相关，那么就可以用长链剖分优化到时空 $O(n)$。唯一要注意的就是 dp 表的位置。

上面这道题是 $dp[\fa][x]$ 依赖 $dp[\son][x-1]$ 的。那如果 $dp[\fa][x]$ 依赖 $dp[\son][x+1]$ 呢？我们要怎么分配空间呢？以 [这道题](https://www.luogu.com.cn/problem/P5904) 为例。

为了解决这道题，我们有一个非常独特的想法。
我们定义，$f[u][j]$ 表示 $u$ 的子树中，到 $u$ 的距离为 $i$ 的节点数；$g[u][i]$ 表示 $u$ 的子树中，若 $x,y\in T_u$，且 $\mathrm{LCA}(x,y)=z$，满足 $d(x,z)=d(y,z)=d(u,z)+i$ 的无序对 $(x,y)$ 个数。

为什么这么定义？我们看下面这张图。
![[Pasted image 20260130015002.png]]

记 $(x,y,z)$ 为满足要求的点，即满足 $a=b+c$，树形 dp 原则是子树内的信息**无后效性**，**尽量把子树内的信息合并在一起**。所以 $a−b=c$，在这个等式种，$a−b$ 和 $c$ 在以 2 为根的两棵不同子树，所以转变统计方式，在 2 号统计答案。**因此上面统计 $i$ 实际上就是在统计 $a-b=i=c$ 的方法数。**

实际上，上面就是在进行分类讨论。
 - 第一类，三个节点位于同一个子树内；
 - 第二类，其中两个位于一个子树，剩下的一个在另一个子树内。

因此，我们可以写出下面的转移方程。假设当前遍历到了 $\newcommand{\from}{\leftarrow}v,x$，从 $x=1..\mathrm{len}[v]$，按照如下方式依次更新 $ans, f, g$：（是的，请注意 dp 不是一下子就做完的我们肯定有中间状态要转移。）
$$
\begin{aligned}
ans&\from ans+f[u][x]\cdot g[v][x+1]+f[v][x-1]\cdot g[u][x],\\
g[u][x]&\from \underbracket{g[v][x+1]}_{\text{from }v}+\underbracket{g[u][x]}_{\text{from }u}+\underbracket{f[u][x]\cdot f[v][x-1]}_{\text{one side from }u,\text{ the other from }v,\ b=0},\\
f[u][x]&\from f[v][x-1]+f[u][x],\\
\end{aligned}
$$

最后别忘了还有 $ans\from ans+g[u][0]$，这是因为有可能上图中 $a=b$，然后我们选择了 $2,x,y$ 三个节点。

这整一个过程都是深度相关的，分别是 $x-1, x+1$，这就用到了下面这个技巧了。

如果 $dp[\fa][x]$ 依赖 $dp[\son][x+1]$，我们这样规划：长链尾的开始位置 = $\dfn[\top]\times 2$，并且从底到顶，开始位置依次加一。

![[Pasted image 20260130013050.png]]

比如上图这样的 dfn。我们分配 $dp[1..26]$。我们模仿 dfs 过程。
 - 我们先处理 $e$。我们从 $id[e]=2\cdot\top[e]=2\times 1=2$ 开始分配 $dp$。$dp[e][0]$ 就填在表中的 $dp[2]$，接下来 $dp[id[e]+k]=dp[e][k]$。
- 接下来在 $d$，我们就分配 $id[d]=2+1=3$ 号编号。
>[!remark] 思考
>为什么这么弄？因为 $dp[\son][i+1]$ 决定了 $dp[\fa][i]$，因此我们这样填就能够天然继承前面的所有 $\son$ 的 dp 信息。

 - 这样走到 $b$，注意到我们现在有另外一个分支了。
	 - 从 $i$ 开始，这是新的长链，从 $2\cdot \top[f]=12$ 开始分配。
	 - 这样，我们腾出了空间，让我们有位置来放置 $a$ 节点的信息。因为理论上我们在这里 $a$ 应该放在 $dp[6]$ 的地方。
	 - 考虑 $dp[a][k]$ 中 $k$ 会去到的位置——我们会取到 $6,7,8,9,10$，因此我们把 $\dfn$ 乘以 2，这样就足以给出一个 $\mathrm{len}$ 的空间。

换言之，我们这里相当于 $dp[i][j]=:dp[id[i]+j]$。

因此，上面这样暴力更新，时间复杂度是 $O(n)$ 的。具体原因上面一道题也说过了。

数组版实现：

```cpp
#include <bits/stdc++.h>

using namespace std;

const int MAXN = 1e5 + 5;

#define int long long

int n;
int fa[MAXN], dfn[MAXN], son[MAXN], len[MAXN], d;
int f[MAXN], g[MAXN << 1], idx[MAXN];
int ans;

int head[MAXN], nxt[MAXN << 1], to[MAXN << 1], ecnt;

void add_edge(int u, int v) {
    nxt[++ecnt] = head[u], head[u] = ecnt, to[ecnt] = v;
}

void setf(int u, int x, int val) { f[dfn[u] + x] = val; }
int getf(int u, int x) { return f[dfn[u] + x]; }
void setg(int u, int x, int val) { g[idx[u] + x] = val; }
int getg(int u, int x) { return g[idx[u] + x]; }

void dfs1(int u, int pa) {
    fa[u] = pa;
    for (int i = head[u]; i; i = nxt[i]) {
        int v = to[i];
        if (v == pa) continue;
        dfs1(v, u);
        if (len[v] > len[son[u]]) son[u] = v;
    }
    len[u] = len[son[u]] + 1;
}

void dfs2(int u, int t) {
    dfn[u] = ++d;
    f[dfn[u]] = 1;
    if (!son[u]) {
        idx[u] = dfn[t] << 1;
        return;
    }
    dfs2(son[u], t);
    idx[u] = idx[son[u]] + 1;
    for (int e = head[u]; e; e = nxt[e]) {
        int v = to[e];
        if (v == fa[u] || v == son[u]) continue;
        dfs2(v, v);
        for (int i = 0; i <= len[v]; i++) {
            if (i) { ans += getg(u, i) * getf(v, i - 1); }
            if (i && i + 1 < len[v]) {
                ans += getf(u, i) * getg(v, i + 1);
            }
            if (i + 1 < len[v]) {
                setg(u, i, getg(u, i) + getg(v, i + 1));
            }
            if (i) {
                setg(u, i, getg(u, i) + getf(u, i) * getf(v, i - 1));
                setf(u, i, getf(u, i) + getf(v, i - 1));
            }
        }
    }
    ans += getg(u, 0);
}

signed main() {
    cin.tie(nullptr)->sync_with_stdio(false);
    cin >> n;
    for (int i = 1; i < n; i++) {
        int u, v;
        cin >> u >> v;
        add_edge(u, v), add_edge(v, u);
    }
    dfs1(1, 0), dfs2(1, 1);
    cout << ans << endl;
    return 0;
}
```

指针版实现：（更加优雅）

```cpp
#include <bits/stdc++.h>
#define N 100005
#define ll long long
using namespace std;
ll tmp[N << 2]; // 只开一个全局
int n;
ll *f[N], *g[N], *id = tmp;
int head[N], idx, dep[N], fa[N], son[N], mx[N];
ll ans = 0;
struct edge {
    int v, next;
} e[2 * N];
void add(int u, int v) {
    e[++idx].v = v;
    e[idx].next = head[u];
    head[u] = idx;
}
void dfs1(int x, int f) {
    fa[x] = f;
    dep[x] = dep[f] + 1;
    for (int i = head[x]; i; i = e[i].next) {
        int y = e[i].v;
        if (y == f) continue;
        dfs1(y, x);
        if (mx[son[x]] < mx[y]) son[x] = y;
    }
    mx[x] = mx[son[x]] + 1;
}
void dfs2(int x) {
    if (son[x]) {
        f[son[x]] = f[x] + 1; // 指针加一
        g[son[x]] = g[x] - 1;
        dfs2(son[x]);
    }
    f[x][0] = 1;
    ans += g[x][0];
    for (int i = head[x]; i; i = e[i].next) {
        int y = e[i].v;
        if (y == fa[x] || y == son[x]) continue;
        f[y] = id;
        id += (mx[y] << 1);
        g[y] = id;
        id += (mx[y] << 1);
        dfs2(y);
        for (int j = 0; j < mx[y]; j++) {
            // 了解指针的话这里可以直接当数组用
            if (j) ans += (f[x][j - 1] * g[y][j]);
            ans += g[x][j + 1] * f[y][j];
        }
        for (int j = 0; j < mx[y]; j++) {
            g[x][j + 1] += f[x][j + 1] * f[y][j];
            if (j) g[x][j - 1] += g[y][j];
            f[x][j + 1] += f[y][j];
        }
    }
}
int main() {
    scanf("%d", &n);
    for (int i = 1; i < n; i++) {
        int x, y;
        scanf("%d%d", &x, &y);
        add(x, y);
        add(y, x);
    }
    dfs1(1, 0);
    f[1] = id;
    id += (mx[1] << 1);
    g[1] = id;
    id += (mx[1] << 1);
    dfs2(1);
    cout << ans << endl;
    return 0;
}
```
