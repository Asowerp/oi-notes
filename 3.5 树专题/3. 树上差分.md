#树 #树上差分

树上差分分为**点差分**和**边差分**。树上差分的一个前置需求是 [[4. 树上倍增·最近公共祖先（LCA）]]。

如果我们要做树上路径修改 + 单点查询，并且不带修，**树上差分**是最优秀的。

# 点差分

树上点差分解决的问题是：给定路径，$a\to b$ 上进行点权的区间修改。我们希望减小修改的时间复杂度，以至于在最后得到正确的点权。

假设我们要修改 $a\to b$ 的路径上的点权 $w_i$，基本逻辑如下：$\newcommand{\LCA}{\operatorname{LCA}}$
1. 修改时，查询 $c=\LCA(a,b)$。$w_a\leftarrow w_a+v$，$w_b\leftarrow w_b+v$，$w_c\leftarrow w_c-v$，$w_{fa[c]}\leftarrow w_c-v$。
2. 查询时，按照如下方式遍历：
	1. `dfs(u)` 来到节点 $u$。遍历 $u$ 的所有子节点 $son$，`dfs(son)`。
	2. 更新 $\displaystyle w_u\leftarrow w_u+\sum_{fa[son]=u}w_{son}$（在遍历的时候不断更新）。
相比暴力每次是 $O(n\log n)$ 的更新，我们直接做到了（因 LCA 查询获得的） $O(\log n)$ 的更新。这是好的。

这样总代价就是 $O(n\log n+q\log n+n)\sim O(n\log n+q\log n)$ 的总时间复杂度。如果使用 tarjan 则是 $O(n+q)$。

为什么上面的操作是合理的呢？思考下面这个路径。
![[Pasted image 20260126154600.png]]

用差分的方式来思考，$\newcommand{\from}{\leftarrow}$$w_a\from w_a+v$ 之后，最后 dfs 更新的时候，从 $a\to lca$  这段路上全都加了 $v$。同理，$w_b\from w_b+v$ 之后，$b\to lca$ 这段路上全都加了 $v$。那 $lca$ 多了一个 $v$，因此需要 $w_{lca}\from w_{lca}-v$。，但是这样 $w_{lca}$ 也多了 $v$，从他往上传递仍然会有问题，因此 $w_{fa[lca]}\from w_{fa[lca]}-v$ 才能够抵消掉这个往上传递的影响。

模板：[P3128 [USACO15DEC] Max Flow P](https://www.luogu.com.cn/problem/P3128)、[P3258 [JLOI2014] 松鼠的新家](https://www.luogu.com.cn/problem/P3258)

```cpp
#include <bits/stdc++.h>

using namespace std;

const int MAXN = 5e4 + 5;

int n, k;
int w[MAXN], pa[20][MAXN], dep[MAXN];
int mx;
vector<int> e[MAXN];

void dfs1(int u, int fa) {
    pa[0][u] = fa;
    dep[u] = dep[fa] + 1;
    for (int i = 1; i <= mx; i++) {
        pa[i][u] = pa[i - 1][pa[i - 1][u]];
    }
    for (auto v : e[u]) {
        if (v == fa) continue;
        dfs1(v, u);
    }
}

int lca(int u, int v) {
    if (dep[u] < dep[v]) swap(u, v);
    for (int i = mx; i >= 0; i--) {
        if (dep[pa[i][u]] >= dep[v]) u = pa[i][u];
    }
    if (u == v) return u;
    for (int i = mx; i >= 0; i--) {
        if (pa[i][u] != pa[i][v]) u = pa[i][u], v = pa[i][v];
    }
    return pa[0][u];
}

void dfs2(int u, int fa) {
    for (auto v : e[u]) {
        if (v == fa) continue;
        dfs2(v, u);
        w[u] += w[v];
    }
}

int main() {
    cin.tie(nullptr)->sync_with_stdio(false);
    cin >> n >> k;
    mx = ceil(log2(n));
    for (int i = 1; i < n; i++) {
        int u, v;
        cin >> u >> v;
        e[u].emplace_back(v), e[v].emplace_back(u);
    }
    dfs1(1, 0);
    for (int i = 1; i <= k; i++) {
        int u, v;
        cin >> u >> v;
        w[u]++, w[v]++;
        int x = lca(u, v);
        w[x]--, w[pa[0][x]]--;
    }
    dfs2(1, 0);
    int ans = 0;
    for (int i = 1; i <= n; i++) { ans = max(ans, w[i]); }
    cout << ans << endl;
    return 0;
}
```

# 边差分

类似，我们想要对边进行区间修改，要怎么做呢？

思考：我们对边直接进行更改是相对麻烦的，我们能不能通过点权更新呢？答案是肯定的。

假设对 $a\to b$ 的路径加 $c$，我们对点权 $w_i$（初始化为 0），边权 $v_i$ 进行如下操作：
1. 修改时，查询 $l=\LCA(a,b)$。$w_a\leftarrow w_a+c$，$w_b\leftarrow w_b+c$，$w_l\leftarrow w_l-2c$。
2. 查询时，按照如下方式进行遍历：
	1. 进行 `dfs(u)`。来到节点 $u$，设 $x$ 是子节点。记 $e=(u,x)$，首先进行 `dfs(x)`。更新 $v_e\from v_e+w_x$。
	2. 更新 $\displaystyle w_u\from w_u+\sum_{fa[x]=u}w_x$（在遍历的时候不断更新）。

正确性类似点差分是容易证明的。

例题：[POJ 3417](http://poj.org/problem?id=3417)：很好的小思考题。我们看一个虚边加上之后实质上控制了一个环，所以转换成边区间加的问题。只要被 2 个及以上虚边控制就无法给出种类数。因此就是板子题了

[P2680 [NOIP 2015 提高组] 运输计划](https://www.luogu.com.cn/problem/P2680)：二分题。我们考虑怎么设计 `check(x)` 函数。定义 `check(x)` 表示在 $x$ 的最大代价下能不能跑完所有的计划。那么所有小于等于 $x$ 的都可以不用考虑。我们接下来考虑大于 $x$ 的计划，加进列表，跑一次边差分（$O(n)$），然后去看有没有一条边满足边权大于 最大值${}-x$，并且被所有边都访问（$O(m)$）。因此总共时间复杂度是 $O((n\log n+m)\log(\sum t_i)+n\log n)$，如果用带 $\log$ 的倍增理论能过，实际不太行。所以改用 tarjan。

这个二分设计确实巧妙。

下面这个是用倍增的，实测过不去。

```cpp
#include <bits/stdc++.h>

using namespace std;

const int MAXN = 300000 + 5;

#define int long long

int n, m;
int head[MAXN], nxt[MAXN << 1], to[MAXN << 1], cost[MAXN << 1],
    from[MAXN << 1];
int e_cnt;
int pa[20][MAXN], path[20][MAXN], dep[MAXN];
int cnt[MAXN];
vector<tuple<int, int, int>> query;

inline void add_edge(int u, int v, int c) {
    nxt[++e_cnt] = head[u], to[e_cnt] = v, head[u] = e_cnt,
    cost[e_cnt] = c, from[e_cnt] = u;
}

inline void dfs1(int u, int fa, int c) {
    pa[0][u] = fa;
    path[0][u] = c;
    dep[u] = dep[fa] + 1;
    for (int i = 1; i <= 19; i++) {
        pa[i][u] = pa[i - 1][pa[i - 1][u]];
        path[i][u] = path[i - 1][pa[i - 1][u]] + path[i - 1][u];
    }
    for (int i = head[u], v; i; i = nxt[i]) {
        v = to[i];
        if (v == fa) continue;
        dfs1(v, u, cost[i]);
    }
}

inline int lca(int u, int v) {
    if (dep[u] < dep[v]) swap(u, v);
    for (int i = 19; i >= 0; i--) {
        if (dep[pa[i][u]] >= dep[v]) u = pa[i][u];
    }
    if (u == v) return u;
    for (int i = 19; i >= 0; i--) {
        if (pa[i][u] != pa[i][v]) u = pa[i][u], v = pa[i][v];
    }
    return pa[0][u];
}

inline int lca_cost(int u, int v) {
    int ret = 0;
    if (dep[u] < dep[v]) swap(u, v);
    for (int i = 19; i >= 0; i--) {
        if (dep[pa[i][u]] >= dep[v]) ret += path[i][u], u = pa[i][u];
    }
    if (u == v) return ret;
    for (int i = 19; i >= 0; i--) {
        if (pa[i][u] != pa[i][v])
            ret += path[i][u], ret += path[i][v], u = pa[i][u],
                                                  v = pa[i][v];
    }
    return ret + path[0][u] + path[0][v];
}

inline void dfs2(int u, int fa) {
    for (int i = head[u], v; i; i = nxt[i]) {
        v = to[i];
        if (v == fa) continue;
        dfs2(v, u);
        cnt[u] += cnt[v];
    }
}

inline bool check(int limit) {
    for (int i = 1; i <= n; i++) cnt[i] = 0;
    int mx = get<0>(query[0]), tmp = 0;
    for (auto [c, u, v] : query) {
        if (c <= limit) break;
        int x = lca(u, v);
        cnt[u]++, cnt[v]++, cnt[x] -= 2;
        tmp++;
    }
    dfs2(1, 0);
    for (int i = 1; i <= e_cnt; i++) {
        if (mx - cost[i] > limit) continue;
        if (cnt[to[i]] == tmp && pa[0][to[i]] == from[i]) return true;
    }
    return false;
}

signed main() {
    cin.tie(nullptr)->sync_with_stdio(false);
    cin >> n >> m;
    int mx = 0, tmp = 0;
    for (int i = 1; i < n; i++) {
        int u, v, c;
        cin >> u >> v >> c;
        add_edge(u, v, c), add_edge(v, u, c);
        mx = max(mx, c);
    }
    dfs1(1, 0, 0);
    for (int i = 1; i <= m; i++) {
        int u, v;
        cin >> u >> v;
        int c = lca_cost(u, v);
        tmp = max(tmp, c);
        query.emplace_back(c, u, v);
    }
    sort(query.begin(), query.end(), greater<>());
    int l = max(0ll, tmp - mx), r = tmp, ans = 0;
    while (l <= r) {
        int mid = l + r >> 1;
        if (check(mid)) {
            r = mid - 1;
            ans = mid;
        } else
            l = mid + 1;
    }
    cout << ans << endl;
    return 0;
}
```

下面这个代码是 tarjan 算法的，请注意我们是怎么维护路径距离的，时间复杂度 $O(m+n+(m+n)\log S)\sim O((m+n)\log S)$：
```cpp
#include <bits/stdc++.h>

#define int long long

using namespace std;

const int MAXN = 3e5 + 5;

int n, m;
int head[MAXN], nxt[MAXN << 1], to[MAXN << 1], from[MAXN << 1],
    cost[MAXN << 1], pa[MAXN], e_cnt, dist[MAXN];
bool vis[MAXN];

void add_edge(int u, int v, int c) {
    nxt[++e_cnt] = head[u], head[u] = e_cnt, to[e_cnt] = v,
    cost[e_cnt] = c, from[e_cnt] = u;
}

class dsu {
    vector<int> fa;

  public:
    dsu() {}
    dsu(int n) : fa(n + 1) { iota(fa.begin(), fa.end(), 0); }
    int find(int x) {
        if (x == fa[x]) { return x; }
        int root = find(fa[x]);
        return fa[x] = root;
    }
    void merge(int x, int y) { fa[x] = y; }
};

dsu uf;
vector<pair<int, int>> q[MAXN];
vector<tuple<int, int, int, int>> query;

void tarjan(int u, int fa, int c) {
    vis[u] = true;
    pa[u] = fa;
    dist[u] = dist[fa] + c;
    for (int i = head[u]; i; i = nxt[i]) {
        int v = to[i];
        if (v == fa) continue;
        tarjan(v, u, cost[i]);
        uf.merge(v, u);
    }
    for (auto [v, id] : q[u]) {
        if (vis[v]) {
            int w = uf.find(v);
            query[id] = {dist[u] + dist[v] - 2 * dist[w], w, u, v};
        }
    }
}

int cnt[MAXN];

void dfs(int u, int fa) {
    for (int i = head[u]; i; i = nxt[i]) {
        int v = to[i];
        if (v == fa) continue;
        dfs(v, u);
        cnt[u] += cnt[v];
    }
}

bool check(int limit) {
    int mx = get<0>(query[0]), tmp = 0;
    for (int i = 1; i <= n; i++) cnt[i] = 0;
    for (auto [c, lca, u, v] : query) {
        if (c <= limit) break;
        tmp++;
        cnt[u]++, cnt[v]++, cnt[lca] -= 2;
    }
    dfs(1, 0);
    for (int i = 1; i <= e_cnt; i++) {
        if (cost[i] < mx - limit || pa[to[i]] != from[i]) continue;
        if (cnt[to[i]] == tmp) return true;
    }
    return false;
}

signed main() {
    cin.tie(nullptr)->sync_with_stdio(false);
    cin >> n >> m;
    int mx0 = 0;
    for (int i = 1; i < n; i++) {
        int u, v, c;
        cin >> u >> v >> c;
        add_edge(u, v, c), add_edge(v, u, c);
        mx0 = max(mx0, c);
    }
    uf = dsu(n);
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        q[u].emplace_back(v, i), q[v].emplace_back(u, i);
        query.emplace_back(u, v, 0, 0);
    }
    tarjan(1, 0, 0);
    sort(query.begin(), query.end(), greater<>());
    int mx = get<0>(query[0]);
    int l = mx - mx0, r = mx, ans = 0;
    while (l <= r) {
        int mid = l + r >> 1;
        if (check(mid)) {
            r = mid - 1;
            ans = mid;
        } else
            l = mid + 1;
    }
    cout << ans << endl;
    return 0;
}
```
